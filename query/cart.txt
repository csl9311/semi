DROP TABLE CART CASCADE CONSTRAINT;
DROP TABLE PAYMENT CASCADE CONSTRAINT;

CREATE SEQUENCE SEQ_CRID START WITH 1;
CREATE TABLE CART(
    CRID NUMBER PRIMARY KEY,
    PID NUMBER,
    PRICE NUMBER,
    AMOUNT NUMBER,
    PNAME VARCHAR2(20),
    POPTION VARCHAR2(100),
    USERID VARCHAR2(20)
);
CREATE TABLE PAYMENT(
   ONO NUMBER,
   USERID VARCHAR2(20 BYTE),
   PHONE VARCHAR2(40 BYTE),
   PID NUMBER NOT NULL ENABLE, 
   PRICE NUMBER NOT NULL ENABLE,
   AMOUNT NUMBER NOT NULL ENABLE,
   PNAME VARCHAR2(100 BYTE) NOT NULL ENABLE, 
   POPTION VARCHAR2(100 BYTE) NOT NULL ENABLE,    
   PAY_DATE DATE NOT NULL ENABLE,
   ADDRESS VARCHAR2(200 BYTE),
   HISTORY VARCHAR2(100 BYTE) NOT NULL ENABLE,
   RECIPIENT VARCHAR2(30 BYTE),
   REQUEST VARCHAR2(200 BYTE),
   
   PRIMARY KEY (ONO,PID)
);

CREATE SEQUENCE SEQ_ONO START WITH 1;
